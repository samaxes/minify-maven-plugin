44  /**
45   * Abstract class for merging and compressing a files list.
46   */
47  public abstract class ProcessFilesTask implements Callable<Object> {
48  
49      protected Log log;
50  
51      protected int linebreak;
52  
53      protected File mergedFile;
54  
55      protected File minifiedFile;
56  
57      protected Integer bufferSize;
58  
59      protected String charset;
60  
61      private List<File> files = new ArrayList<File>();
62  
63      /**
64       * Task constructor.
65       *
66       * @param log Maven plugin log
67       * @param bufferSize size of the buffer used to read source files.
68       * @param webappSourceDir web resources source directory
69       * @param webappTargetDir web resources target directory
70       * @param inputDir directory containing source files
71       * @param sourceFiles list of source files to include
72       * @param sourceIncludes list of source files to include
73       * @param sourceExcludes list of source files to exclude
74       * @param outputDir directory to write the final file
75       * @param finalFilename final filename
76       * @param suffix final filename suffix
77       * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
78       *        Otherwise, only byte-to-byte operations are used
79       * @param linebreak split long lines after a specific column
80       */
81      public ProcessFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
82              String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
83              String outputDir, String finalFilename, String suffix, String charset, int linebreak) {
84          this.log = log;
85          this.linebreak = linebreak;
86          this.bufferSize = bufferSize;
87          this.charset = charset;
88  
89          File sourceDir = new File(webappSourceDir + File.separator + inputDir);
90          for (String sourceFilename : sourceFiles) {
91              addNewSourceFile(finalFilename, sourceDir, sourceFilename);
92          }
93          for (File sourceInclude : getFilesToInclude(sourceDir, sourceIncludes, sourceExcludes)) {
94              if (!files.contains(sourceInclude)) {
95                  addNewSourceFile(finalFilename, sourceDir, sourceInclude.getName());
96              }
97          }
98  
99          File targetDir = new File(webappTargetDir + File.separator + outputDir);
100         String extension = "." + FileUtils.getExtension(finalFilename);
101         if (!files.isEmpty() && (targetDir.exists() || targetDir.mkdirs())) {
102             this.mergedFile = new File(targetDir, finalFilename);
103             this.minifiedFile = new File(targetDir, this.mergedFile.getName().replace(extension,
104                     suffix + extension));
105         } else if (!sourceFiles.isEmpty() || !sourceIncludes.isEmpty()) {
106             // The 'files' list will be empty if the source file paths or names added to the project's POM are wrong.
107             String fileType = ("CSS".equalsIgnoreCase(extension.substring(1))) ? "CSS" : "JavaScript";
108             log.error("No " + fileType + " source files found to process.");
109         }
110     }

112     /**
113      * Method executed by the thread.
114      */
115     public Object call() {
116         mergeFiles();
117         minify();
118 
119         return null;
120     }

122     /**
123      * Logs an addition of a new source file.
124      *
125      * @param finalFilename the final file name
126      * @param sourceDir the sources directory
127      * @param sourceFilename the source file name
128      */
129     private void addNewSourceFile(String finalFilename, File sourceDir, String sourceFilename) {
130         File sourceFile = new File(sourceDir, sourceFilename);
131 
132         addNewSourceFile(finalFilename, sourceDir, sourceFile);
133     }
134 
135     /**
136      * Logs an addition of a new source file.
137      *
138      * @param finalFilename the final file name
139      * @param sourceDir the sources directory
140      * @param sourceFile the source file
141      */
142     private void addNewSourceFile(String finalFilename, File sourceDir, File sourceFile) {
143         if (sourceFile.exists()) {
144             if (finalFilename.equalsIgnoreCase(sourceFile.getName())) {
145                 log.warn("Source file [" + sourceFile.getName() + "] has the same name as the final file.");
146             }
147             log.debug("Source file [" + sourceFile.getName() + "] added.");
148             files.add(sourceFile);
149         } else {
150             log.warn("Source file [" + sourceFile.getName() + "] was not included beacause it does not exist.");
151         }
152     }
153 
154     /**
155      * Returns the files to copy. Default exclusions are used when the excludes list is empty.
156      *
157      * @param baseDir the base directory to start from
158      * @param includes list of source files to include
159      * @param excludes list of source files to exclude
160      * @return the files to copy
161      */
162     private List<File> getFilesToInclude(File baseDir, List<String> includes, List<String> excludes) {
163         List<File> includedFiles = new ArrayList<File>();
164 
165         if (includes != null && !includes.isEmpty()) {
166             DirectoryScanner scanner = new DirectoryScanner();
167 
168             scanner.setIncludes(includes.toArray(new String[0]));
169             scanner.setExcludes(excludes.toArray(new String[0]));
170             scanner.addDefaultExcludes();
171             scanner.setBasedir(baseDir);
172             scanner.scan();
173 
174             for (String includedFilename : scanner.getIncludedFiles()) {
175                 includedFiles.add(new File(baseDir, includedFilename));
176             }
177 
178             Collections.sort(includedFiles, new FilenameComparator());
179         }
180 
181         return includedFiles;
182     }
183 
184     /**
185      * Merges files list.
186      */
187     private void mergeFiles() {
188         if (mergedFile != null) {
189             ListOfFiles listOfFiles = new ListOfFiles(log, files);
190 
191             try {
192                 log.info("Creating merged file [" + mergedFile.getName() + "].");
193                 InputStream sequence = new SequenceInputStream(listOfFiles);
194                 OutputStream out = new FileOutputStream(mergedFile);
195 
196                 if (charset == null) {
197                     IOUtil.copy(sequence, out, bufferSize);
198                 } else {
199                     InputStreamReader sequenceReader = new InputStreamReader(sequence, charset);
200                     OutputStreamWriter outWriter = new OutputStreamWriter(out, charset);
201 
202                     IOUtil.copy(sequenceReader, outWriter, bufferSize);
203                     IOUtil.close(sequenceReader);
204                     IOUtil.close(outWriter);
205                 }
206 
207                 IOUtil.close(sequence);
208                 IOUtil.close(out);
209             } catch (IOException e) {
210                 log.error("An error has occurred while concatenating files.", e);
211             }
212         }
213     }
