30  /**
31   * <a href="https://developers.google.com/closure/compiler/">Google Closure Compiler</a> configuration.
32   */
33  public class ClosureConfig {
34  
35      private final LanguageMode languageIn;
36  
37      private final LanguageMode languageOut;
38  
39      private final CompilerOptions.Environment environment;
40  
41      private final CompilationLevel compilationLevel;
42  
43      private final DependencyOptions dependencyOptions;
44  
45      private final List<SourceFile> externs;
46  
47      private final Format sourceMapFormat;
48  
49      private final Map<DiagnosticGroup, CheckLevel> warningLevels;
50  
51      private final Boolean colorizeErrorOutput;
52  
53      private final Boolean angularPass;
54  
55      private final List<String> extraAnnotations;
56  
57      private final Map<String, Object> defineReplacements = new HashMap<>();
58  
59      /**
60       * Init Closure Compiler values.
61       *
62       * @param languageIn         the version of ECMAScript used to report errors in the code
63       * @param languageOut        the version of ECMAScript the code will be returned in
64       * @param environment        the set of builtin externs to load
65       * @param compilationLevel   the degree of compression and optimization to apply to JavaScript
66       * @param dependencyOptions  options for how to manage dependencies between input files
67       * @param externs            preserve symbols that are defined outside of the code you are compiling
68       * @param createSourceMap    create a source map for the minifed/combined production files
69       * @param warningLevels      a map of warnings to enable or disable in the compiler
70       * @param angularPass        use {@code @ngInject} annotation to generate Angular injections
71       * @param extraAnnotations   make extra annotations known to the closure engine
72       * @param defineReplacements replacements for {@code @defines}
73       */
74      public ClosureConfig(LanguageMode languageIn, LanguageMode languageOut, CompilerOptions.Environment environment,
75                           CompilationLevel compilationLevel, DependencyOptions dependencyOptions,
76                           List<SourceFile> externs, boolean createSourceMap,
77                           Map<DiagnosticGroup, CheckLevel> warningLevels, boolean angularPass,
78                           List<String> extraAnnotations, Map<String, String> defineReplacements) {
79          this.languageIn = languageIn;
80          this.languageOut = languageOut;
81          this.environment = environment;
82          this.compilationLevel = compilationLevel;
83          this.dependencyOptions = dependencyOptions;
84          this.externs = externs;
85          this.sourceMapFormat = (createSourceMap) ? SourceMap.Format.V3 : null;
86          this.warningLevels = warningLevels;
87          this.colorizeErrorOutput = Boolean.TRUE;
88          this.angularPass = angularPass;
89          this.extraAnnotations = extraAnnotations;
90  
91          for (Map.Entry<String, String> defineReplacement : defineReplacements.entrySet()) {
92              if (Strings.isNullOrEmpty(defineReplacement.getValue())) {
93                  throw new RuntimeException("Define replacement " + defineReplacement.getKey() + " does not have a value.");
94              }
95  
96              if (String.valueOf(true).equals(defineReplacement.getValue()) ||
97                      String.valueOf(false).equals(defineReplacement.getValue())) {
98                  this.defineReplacements.put(defineReplacement.getKey(), Boolean.valueOf(defineReplacement.getValue()));
99                  continue;
100             }
101 
102             try {
103                 this.defineReplacements.put(defineReplacement.getKey(), Integer.valueOf(defineReplacement.getValue()));
104                 continue;
105             } catch (NumberFormatException e) {
106                 // Not a valid Integer, try next type
107             }
108 
109             try {
110                 this.defineReplacements.put(defineReplacement.getKey(), Double.valueOf(defineReplacement.getValue()));
111                 continue;
112             } catch (NumberFormatException e) {
113                 // Not a valid Double, try next type
114             }
115 
116             this.defineReplacements.put(defineReplacement.getKey(), defineReplacement.getValue());
117         }
118     }
