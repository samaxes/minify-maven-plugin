39  /**
40   * Task for merging and compressing JavaScript files.
41   */
42  public class ProcessJSFilesTask extends ProcessFilesTask {
43  
44      private boolean munge;
45  
46      private boolean verbose;
47  
48      private boolean preserveAllSemiColons;
49  
50      private boolean disableOptimizations;
51  
52      /**
53       * Task constructor.
54       *
55       * @param log Maven plugin log
56       * @param bufferSize size of the buffer used to read source files
57       * @param debug show source file paths in log output
58       * @param skipMerge whether to skip the merge step or not
59       * @param skipMinify whether to skip the minify step or not
60       * @param webappSourceDir web resources source directory
61       * @param webappTargetDir web resources target directory
62       * @param inputDir directory containing source files
63       * @param sourceFiles list of source files to include
64       * @param sourceIncludes list of source files to include
65       * @param sourceExcludes list of source files to exclude
66       * @param outputDir directory to write the final file
67       * @param outputFilename the output file name
68       * @param suffix final filename suffix
69       * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
70       *        Otherwise, only byte-to-byte operations are used
71       * @param linebreak split long lines after a specific column
72       * @param munge minify only
73       * @param verbose display informational messages and warnings
74       * @param preserveAllSemiColons preserve unnecessary semicolons
75       * @param disableOptimizations disable all the built-in micro optimizations
76       */
77      public ProcessJSFilesTask(Log log, Integer bufferSize, boolean debug, boolean skipMerge, boolean skipMinify,
78              String webappSourceDir, String webappTargetDir, String inputDir, List<String> sourceFiles,
79              List<String> sourceIncludes, List<String> sourceExcludes, String outputDir, String outputFilename,
80              String suffix, String charset, int linebreak, boolean munge, boolean verbose,
81              boolean preserveAllSemiColons, boolean disableOptimizations) {
82          super(log, bufferSize, debug, skipMerge, skipMinify, webappSourceDir, webappTargetDir, inputDir, sourceFiles,
83                  sourceIncludes, sourceExcludes, outputDir, outputFilename, suffix, charset, linebreak);
84  
85          this.munge = munge;
86          this.verbose = verbose;
87          this.preserveAllSemiColons = preserveAllSemiColons;
88          this.disableOptimizations = disableOptimizations;
89      }
90  
91      /**
92       * Minifies JavaScript file.
93       *
94       * @param mergedFile input file resulting from the merged step
95       * @param minifiedFile output file resulting from the minify step
96       */
97      @Override
98      protected void minify(File mergedFile, File minifiedFile) {
99          if (minifiedFile != null) {
100             try {
101                 log.info("Creating minified file [" + ((debug) ? minifiedFile.getPath() : minifiedFile.getName())
102                         + "].");
103 
104                 InputStream in = new FileInputStream(mergedFile);
105                 OutputStream out = new FileOutputStream(minifiedFile);
106                 InputStreamReader reader;
107                 OutputStreamWriter writer;
108                 if (charset == null) {
109                     reader = new InputStreamReader(in);
110                     writer = new OutputStreamWriter(out);
111                 } else {
112                     reader = new InputStreamReader(in, charset);
113                     writer = new OutputStreamWriter(out, charset);
114                 }
115 
116                 JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
117                         mergedFile.getName()));
118                 compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
119 
120                 IOUtil.close(reader);
121                 IOUtil.close(writer);
122                 IOUtil.close(in);
123                 IOUtil.close(out);
124             } catch (IOException e) {
125                 log.error(e.getMessage(), e);
126             }
127         }
128     }
