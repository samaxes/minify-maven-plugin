21  package com.samaxes.maven.plugin.minify;
22  
23  import java.io.FileReader;
24  import java.io.FileWriter;
25  import java.io.IOException;
26  import java.io.Reader;
27  import java.io.Writer;
28  
29  import org.apache.maven.plugin.logging.Log;
30  
31  import com.yahoo.platform.yui.compressor.CssCompressor;
32  import com.yahoo.platform.yui.compressor.JavaScriptCompressor;
33  
34  /**
35   * Task for compressing a file.
36   */
37  public class CompressFileTask implements Runnable {
38  
39      private Log log;
40  
41      private String extension;
42  
43      private String inputPathname;
44  
45      private String outputPathname;
46  
47      private int linebreak;
48  
49      private boolean munge;
50  
51      private boolean verbose;
52  
53      private boolean preserveAllSemiColons;
54  
55      private boolean disableOptimizations;
56  
57      private final String suffix = ".min";
58  
59      /**
60       * Task constructor.
61       * 
62       * @param log Maven plugin log
63       * @param inputPathname the input pathname
64       * @param linebreak split long lines after a specific column
65       * @param munge minify only
66       * @param verbose display informational messages and warnings
67       * @param preserveAllSemiColons preserve unnecessary semicolons
68       * @param disableOptimizations disable all the built-in micro optimizations
69       */
70      public CompressFileTask(Log log, String inputPathname, int linebreak, boolean munge, boolean verbose,
71              boolean preserveAllSemiColons, boolean disableOptimizations) {
72          this.log = log;
73          this.extension = inputPathname.substring(inputPathname.lastIndexOf('.'));
74          this.inputPathname = inputPathname;
75          this.outputPathname = inputPathname.replace(extension, suffix.concat(extension));
76          this.linebreak = linebreak;
77          this.munge = munge;
78          this.verbose = verbose;
79          this.preserveAllSemiColons = preserveAllSemiColons;
80          this.disableOptimizations = disableOptimizations;
81      }
82  
83      /**
84       * Method executed by the thread.
85       */
86      public void run() {
87          try {
88              Reader reader = new FileReader(inputPathname);
89              Writer writer = new FileWriter(outputPathname);
90  
91              if (".css".equalsIgnoreCase(extension)) {
92                  CssCompressor compressor = new CssCompressor(reader);
93                  compressor.compress(writer, linebreak);
94              } else if (".js".equalsIgnoreCase(extension)) {
95                  JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log));
96                  compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
97              }
98  
99              reader.close();
100             writer.close();
101         } catch (IOException e) {
102             log.error(e.getMessage(), e);
103         }
104     }
105 }
