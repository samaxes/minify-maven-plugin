21  package com.samaxes.maven.plugin.minify;
22  
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.concurrent.ExecutionException;
26  import java.util.concurrent.ExecutorService;
27  import java.util.concurrent.Executors;
28  import java.util.concurrent.Future;
29  
30  import org.apache.maven.plugin.AbstractMojo;
31  import org.apache.maven.plugin.MojoExecutionException;
32  import org.apache.maven.plugin.MojoFailureException;

34  /**
35   * Goal which combines and minifies CSS and JavaScript files.
36   * 
37   * @goal minify
38   * @phase process-resources
39   */
40  public class MinifyMojo extends AbstractMojo {
41  
42      /**
43       * Webapp source directory.
44       * 
45       * @parameter expression="${minify.webappSourceDir}" default-value="${basedir}/src/main/webapp"
46       * @required
47       */
48      private String webappSourceDir;
49  
50      /**
51       * Webapp target directory.
52       * 
53       * @parameter expression="${minify.webappTargetDir}"
54       *            default-value="${project.build.directory}/${project.build.finalName}"
55       * @required
56       */
57      private String webappTargetDir;
58  
59      /**
60       * CSS source directory.
61       * 
62       * @parameter expression="${minify.cssDir}" default-value="css"
63       * @required
64       */
65      private String cssDir;
66  
67      /**
68       * JavaScript source directory.
69       * 
70       * @parameter expression="${minify.jsDir}" default-value="js"
71       * @required
72       */
73      private String jsDir;
74  
75      /**
76       * CSS filenames list.
77       * 
78       * @parameter
79       */
80      private List<String> cssFiles = new ArrayList<String>();
81  
82      /**
83       * JavaScript filenames list.
84       * 
85       * @parameter
86       */
87      private List<String> jsFiles = new ArrayList<String>();
88  
89      /**
90       * CSS output filename.
91       * 
92       * @parameter expression="${minify.cssFinalFile}" default-value="style.css"
93       */
94      private String cssFinalFile;
95  
96      /**
97       * CSS file extension.
98       * 
99       * @parameter expression="${minify.cssExtension}" default-value=".css"
100      */
101     private String cssExtension;
102 
103     /**
104      * JavaScript output filename.
105      * 
106      * @parameter expression="${minify.jsFinalFile}" default-value="script.js"
107      */
108     private String jsFinalFile;
109 
110     /**
111      * JavaScript file extension.
112      * 
113      * @parameter expression="${minify.jsExtension}" default-value=".js"
114      */
115     private String jsExtension;
116 
117     /**
118      * Some source control tools don't like files containing lines longer than, say 8000 characters. The linebreak
119      * option is used in that case to split long lines after a specific column. It can also be used to make the code
120      * more readable, easier to debug (especially with the MS Script Debugger). Specify 0 to get a line break after each
121      * semi-colon in JavaScript, and after each rule in CSS. Specify -1 to disallow line breaks.
122      * 
123      * @parameter expression="${minify.linebreak}" default-value="-1"
124      * @required
125      */
126     private int linebreak;
127 
128     /**
129      * Minify only. Do not obfuscate local symbols.
130      * 
131      * @parameter expression="${minify.munge}" default-value="false"
132      * @required
133      */
134     private boolean nomunge;
135 
136     /**
137      * Display informational messages and warnings.
138      * 
139      * @parameter expression="${minify.verbose}" default-value="false"
140      * @required
141      */
142     private boolean verbose;
143 
144     /**
145      * Preserve unnecessary semicolons (such as right before a '}'). This option is useful when compressed code has to
146      * be run through JSLint (which is the case of YUI for example).
147      * 
148      * @parameter expression="${minify.preserveAllSemiColons}" default-value="false"
149      * @required
150      */
151     private boolean preserveAllSemiColons;
152 
153     /**
154      * Disable all the built-in micro optimizations.
155      * 
156      * @parameter expression="${minify.disableOptimizations}" default-value="false"
157      * @required
158      */
159     private boolean disableOptimizations;
160 
161     /**
162      * Size of the buffer used to read source files.
163      * 
164      * @parameter expression="${minify.bufferSize}" default-value="4096"
165      * @required
166      */
167     private int bufferSize;
168 
169     /**
170      * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
171      */
172     public void execute() throws MojoExecutionException, MojoFailureException {
173         ExecutorService executor = Executors.newFixedThreadPool(2);
174         Future<?> processCSSFilesTask = (cssFiles.isEmpty()) ? null : executor.submit(new ProcessFilesTask(getLog(),
175                 bufferSize, webappSourceDir, webappTargetDir, cssDir, cssFiles, cssFinalFile, cssExtension, linebreak,
176                 !nomunge, verbose, preserveAllSemiColons, disableOptimizations));
177         Future<?> processJSFilesTask = (jsFiles.isEmpty()) ? null : executor.submit(new ProcessFilesTask(getLog(),
178                 bufferSize, webappSourceDir, webappTargetDir, jsDir, jsFiles, jsFinalFile, jsExtension, linebreak,
179                 !nomunge, verbose, preserveAllSemiColons, disableOptimizations));
180 
181         try {
182             if (processCSSFilesTask != null) {
183                 processCSSFilesTask.get();
184             }
185             if (processJSFilesTask != null) {
186                 processJSFilesTask.get();
187             }
188         } catch (InterruptedException e) {
189             getLog().error(e.getMessage(), e);
190         } catch (ExecutionException e) {
191             getLog().error(e.getMessage(), e);
192         }
193     }
