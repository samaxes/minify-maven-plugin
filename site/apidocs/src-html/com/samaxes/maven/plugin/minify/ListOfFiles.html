021    package com.samaxes.maven.plugin.minify;
022    
023    import java.io.FileInputStream;
024    import java.io.FileNotFoundException;
025    import java.io.InputStream;
026    import java.util.Enumeration;
027    import java.util.List;
028    import java.util.NoSuchElementException;
029    
030    import org.apache.maven.plugin.logging.Log;
031    
032    /**
033     * {@code ListOfFiles} is used to initialize the SequenceInputStream which uses {@code ListOfFiles} to get a new
034     * InputStream for every filename listed.
035     */
036    public class ListOfFiles implements Enumeration<InputStream> {
037    
038        private Log log;
039    
040        private List<String> listOfFiles;
041    
042        private int current = 0;
043    
044        /**
045         * ListOfFiles public constructor.
046         * 
047         * @param log Maven plugin log
048         * @param listOfFiles list of pathnames
049         */
050        public ListOfFiles(Log log, List<String> listOfFiles) {
051            this.log = log;
052            this.listOfFiles = listOfFiles;
053        }
054    
055        /**
056         * Tests if this enumeration contains more elements.
057         * 
058         * @return <code>true</code> if and only if this enumeration object contains at least one more element to provide;
059         *         <code>false</code> otherwise.
060         */
061        public boolean hasMoreElements() {
062            return (current < listOfFiles.size()) ? true : false;
063        }
064    
065        /**
066         * Returns the next element of this enumeration if this enumeration object has at least one more element to provide.
067         * 
068         * @return the next element of this enumeration.
069         * @exception NoSuchElementException if no more elements exist.
070         */
071        public InputStream nextElement() {
072            InputStream is = null;
073    
074            if (!hasMoreElements())
075                throw new NoSuchElementException("No more files.");
076            else {
077                try {
078                    String nextElement = listOfFiles.get(current);
079                    current++;
080                    is = new FileInputStream(nextElement);
081                } catch (FileNotFoundException e) {
082                    log.error("The file " + listOfFiles.get(current) + " was not found.", e);
083                }
084            }
085    
086            return is;
087        }
088    }
