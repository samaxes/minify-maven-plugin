021package com.samaxes.maven.plugin.minify;
022
023import java.io.FileInputStream;
024import java.io.FileOutputStream;
025import java.io.IOException;
026import java.io.InputStream;
027import java.io.InputStreamReader;
028import java.io.OutputStream;
029import java.io.OutputStreamWriter;
030import java.util.List;
031
032import org.apache.maven.plugin.logging.Log;
033import org.codehaus.plexus.util.IOUtil;
034
035import com.samaxes.maven.plugin.common.JavaScriptErrorReporter;
036import com.yahoo.platform.yui.compressor.JavaScriptCompressor;
037
038/**
039 * Task for merging and compressing JavaScript files.
040 */
041public class ProcessJSFilesTask extends ProcessFilesTask {
042
043    private boolean munge;
044
045    private boolean verbose;
046
047    private boolean preserveAllSemiColons;
048
049    private boolean disableOptimizations;
050
051    /**
052     * Task constructor.
053     *
054     * @param log Maven plugin log
055     * @param bufferSize size of the buffer used to read source files.
056     * @param webappSourceDir web resources source directory
057     * @param webappTargetDir web resources target directory
058     * @param inputDir directory containing source files
059     * @param sourceFiles list of source files to include
060     * @param sourceIncludes list of source files to include
061     * @param sourceExcludes list of source files to exclude
062     * @param outputDir directory to write the final file
063     * @param finalFilename final filename
064     * @param suffix final filename suffix
065     * @param charset if a character set is specified, a byte-to-char variant allows the encoding to be selected.
066     *        Otherwise, only byte-to-byte operations are used
067     * @param linebreak split long lines after a specific column
068     * @param munge minify only
069     * @param verbose display informational messages and warnings
070     * @param preserveAllSemiColons preserve unnecessary semicolons
071     * @param disableOptimizations disable all the built-in micro optimizations
072     */
073    public ProcessJSFilesTask(Log log, Integer bufferSize, String webappSourceDir, String webappTargetDir,
074            String inputDir, List<String> sourceFiles, List<String> sourceIncludes, List<String> sourceExcludes,
075            String outputDir, String finalFilename, String suffix, String charset, int linebreak, boolean munge,
076            boolean verbose, boolean preserveAllSemiColons, boolean disableOptimizations) {
077        super(log, bufferSize, webappSourceDir, webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes,
078                outputDir, finalFilename, suffix, charset, linebreak);
079
080        this.munge = munge;
081        this.verbose = verbose;
082        this.preserveAllSemiColons = preserveAllSemiColons;
083        this.disableOptimizations = disableOptimizations;
084    }
085
086    /**
087     * Minifies JavaScript file.
088     */
089    @Override
090    protected void minify() {
091        if (minifiedFile != null) {
092            try {
093                log.info("Creating minified file [" + minifiedFile.getName() + "].");
094
095                InputStream in = new FileInputStream(mergedFile);
096                OutputStream out = new FileOutputStream(minifiedFile);
097                InputStreamReader reader;
098                OutputStreamWriter writer;
099                if (charset == null) {
100                    reader = new InputStreamReader(in);
101                    writer = new OutputStreamWriter(out);
102                } else {
103                    reader = new InputStreamReader(in, charset);
104                    writer = new OutputStreamWriter(out, charset);
105                }
106
107                JavaScriptCompressor compressor = new JavaScriptCompressor(reader, new JavaScriptErrorReporter(log,
108                        mergedFile.getName()));
109                compressor.compress(writer, linebreak, munge, verbose, preserveAllSemiColons, disableOptimizations);
110
111                IOUtil.close(reader);
112                IOUtil.close(writer);
113                IOUtil.close(in);
114                IOUtil.close(out);
115            } catch (IOException e) {
116                log.error(e.getMessage(), e);
117            }
118        }
119    }
120}
