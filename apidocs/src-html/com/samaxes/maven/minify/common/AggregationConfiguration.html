019package com.samaxes.maven.minify.common;
020
021import java.util.List;
022
023/**
024 * Maps aggregations from an external JSON formatted config file defined in the option {@code bundleConfiguration}.
025 */
026public class AggregationConfiguration {
027
028    private List<Aggregation> bundles;
029
030    /**
031     * Gets the bundles.
032     *
033     * @return the bundles
034     */
035    public List<Aggregation> getBundles() {
036        return bundles;
037    }
038
039    /**
040     * Sets the bundles.
041     *
042     * @param bundles the bundles to set
043     */
044    public void setBundles(List<Aggregation> bundles) {
045        this.bundles = bundles;
046    }
047}
