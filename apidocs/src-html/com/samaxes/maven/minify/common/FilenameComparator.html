019package com.samaxes.maven.minify.common;
020
021import java.io.File;
022import java.util.Comparator;
023
024/**
025 * Custom file name comparator. Compares file name instead of file path.
026 */
027public class FilenameComparator implements Comparator<File> {
028
029    /**
030     * Compares two file names lexicographically, ignoring case differences. This method returns an integer whose sign is
031     * that of calling compareTo with normalized versions of the strings where case differences have been eliminated by
032     * calling Character.toLowerCase(Character.toUpperCase(character)) on each character.
033     *
034     * @param o1 The first file object to be compared
035     * @param o2 The second file object to be compared
036     * @return Zero if both the files have the same name, a value less than zero if the first file is lexicographically
037     *         less than the second, or a value greater than zero if the first file is lexicographically greater than
038     *         the second
039     */
040    @Override
041    public int compare(File o1, File o2) {
042        return o1.getName().compareToIgnoreCase(o2.getName());
043    }
044}
