021package com.samaxes.maven.minify.common;
022
023import org.apache.maven.plugin.logging.Log;
024import org.mozilla.javascript.ErrorReporter;
025import org.mozilla.javascript.EvaluatorException;
026
027/**
028 * Reports any error occurring during JavaScript files compression.
029 */
030public class JavaScriptErrorReporter implements ErrorReporter {
031
032    private Log log;
033
034    private String filename;
035
036    /**
037     * Error reporter constructor.
038     *
039     * @param log Maven plugin log
040     * @param filename JavaScript source filename
041     */
042    public JavaScriptErrorReporter(Log log, String filename) {
043        this.log = log;
044        this.filename = filename;
045    }
046
047    /**
048     * Reports a warning.
049     *
050     * @param message a String describing the warning
051     * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
052     *        URL
053     * @param line the line number associated with the warning
054     * @param lineSource the text of the line (may be null)
055     * @param lineOffset the offset into lineSource where problem was detected
056     */
057    @Override
058    public void warning(String message, String sourceName, int line, String lineSource, int lineOffset) {
059        if (line < 0) {
060            log.warn(message);
061        } else {
062            log.warn("[" + filename + ":" + line + "] " + message);
063        }
064    }
065
066    /**
067     * Reports an error. If execution has not yet begun, the JavaScript engine is free to find additional errors rather
068     * than terminating the translation. However, it will not execute a script that had errors.
069     *
070     * @param message a String describing the warning
071     * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
072     *        URL
073     * @param line the line number associated with the warning
074     * @param lineSource the text of the line (may be null)
075     * @param lineOffset the offset into lineSource where problem was detected
076     */
077    @Override
078    public void error(String message, String sourceName, int line, String lineSource, int lineOffset) {
079        if (line < 0) {
080            log.error(message);
081        } else {
082            log.error("[" + filename + ":" + line + "] " + message);
083        }
084    }
085
086    /**
087     * Creates an EvaluatorException that may be thrown. runtimeErrors, unlike errors, will always terminate the current
088     * script.
089     *
090     * @param message a String describing the warning
091     * @param sourceName a String describing the JavaScript source where the warning occured; typically a filename or
092     *        URL
093     * @param line the line number associated with the warning
094     * @param lineSource the text of the line (may be null)
095     * @param lineOffset the offset into lineSource where problem was detected
096     */
097    @Override
098    public EvaluatorException runtimeError(String message, String sourceName, int line, String lineSource,
099            int lineOffset) {
100        error(message, sourceName, line, lineSource, lineOffset);
101
102        return new EvaluatorException(message);
103    }
104}
